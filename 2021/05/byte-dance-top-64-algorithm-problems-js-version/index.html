        if (str === '(') {

        }
    }

class Stack {
    constructor() {
        this.count = 0;
        this.items = [];
    }

    push(element) {
        this.items[this.count] = element;
        this.count++;
    }

    pop() {
        if (this.isEmpty()) return;
        const element = this.items[this.count - 1];
        delete this.items[this.count - 1];
        this.count--;
        return element;
    }

    size() {
        return this.count;
    }

    isEmpty() {
        return this.size() === 0;
    }
}
